import { useState } from 'react'
import {
  Stepper,
  Step,
  StepLabel,
  Button,
  Box,
  Typography
} from 'src/components/atoms'
import { Meta, Story, Preview } from '@storybook/addon-docs/blocks'

<Meta title="Components/atoms/Stepper" component={Stepper} />

# Stepper

This component was created from Stepper of [@material-ui](https://material-ui.com).

[Stepper API](https://material-ui.com/api/stepper)

<Preview>
  <Story name="Stepper">
    {() => {
      const [step, setStep] = useState(0)
      const steps = ['First Step', 'Second Step', 'Third Step']
      const handleMin = () => {
        if (step > 0) {
          setStep(step - 1)
        }
      }
      const handleMax = () => {
        if (step < steps.length) {
          setStep(step + 1)
        }
      }
      return (
        <>
          <Typography> Horizontal </Typography>
          <Stepper alternativeLabel activeStep={step}>
            {steps.map((stepTitle, index) => (
              <Step key={index}>
                <StepLabel>{stepTitle}</StepLabel>
              </Step>
            ))}
          </Stepper>
          <Box display="flex" justifyContent="space-around">
            <Button
              onClick={handleMin}
              variant="contained"
              color="secondary"
              size="small"
            >
              Back
            </Button>
            <Button
              onClick={handleMax}
              variant="contained"
              color="secondary"
              size="small"
            >
              Next
            </Button>
          </Box>
          <Typography> Horizontal </Typography>
          <Stepper orientation="vertical" activeStep={step}>
            {steps.map((stepTitle, index) => (
              <Step key={index}>
                <StepLabel>{stepTitle}</StepLabel>
              </Step>
            ))}
          </Stepper>
        </>
      )
    }}
  </Story>
</Preview>
